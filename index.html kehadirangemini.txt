/**
 * GOOGLE APPS SCRIPT BACKEND
 * Fungsi ini bertindak sebagai API untuk Frontend.
 */

const SS_NAME = "Database_Absensi_Sekolah"; // Nama file spreadsheet yang akan dibuat

/**
 * 1. FUNGSI INISIALISASI OTOMATIS
 * Membuat Spreadsheet dan semua Sheet yang dibutuhkan jika belum ada.
 */
function initDatabase() {
  let files = DriveApp.getFilesByName(SS_NAME);
  let ss;
  
  if (files.hasNext()) {
    ss = SpreadsheetApp.open(files.next());
  } else {
    ss = SpreadsheetApp.create(SS_NAME);
  }
  
  const sheetsConfig = {
    "Murid": ["ID Murid", "Nama", "Kelas", "Status Aktif"],
    "Kelas": ["ID Kelas", "Nama Kelas", "Jumlah Murid", "ID Wali Kelas"],
    "Wali_Kelas": ["ID Wali", "Nama", "NIP", "Username", "Password"],
    "Absensi": ["ID Absen", "ID Murid", "Tanggal", "Jam", "Status", "Keterangan", "ID Wali Kelas"],
    "Hari_Libur": ["ID Libur", "Tanggal", "Keterangan"],
    "Admin": ["ID Admin", "Username", "Password"],
    "Pengaturan": ["Nama Sekolah", "Nama Kepala Sekolah", "NIP Kepala Sekolah"]
  };
  
  for (let name in sheetsConfig) {
    let sheet = ss.getSheetByName(name);
    if (!sheet) {
      sheet = ss.insertSheet(name);
      sheet.appendRow(sheetsConfig[name]);
      // Format Header Bold
      sheet.getRange(1, 1, 1, sheetsConfig[name].length).setFontWeight("bold").setBackground("#f3f3f3");
    }
  }
  
  // Hapus sheet default "Sheet1" jika ada
  let defaultSheet = ss.getSheetByName("Sheet1");
  if (defaultSheet) ss.deleteSheet(defaultSheet);
  
  return { status: "success", id: ss.getId() };
}

/**
 * 2. HANDLING LOGIN
 * Mengecek data di sheet Admin atau Wali_Kelas
 */
function checkLogin(username, password) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  
  // Cek di Admin
  const adminData = ss.getSheetByName("Admin").getDataRange().getValues();
  for (let i = 1; i < adminData.length; i++) {
    if (adminData[i][1] === username && adminData[i][2] === password) {
      return { role: 'admin', name: username };
    }
  }
  
  // Cek di Wali Kelas
  const waliData = ss.getSheetByName("Wali_Kelas").getDataRange().getValues();
  for (let i = 1; i < waliData.length; i++) {
    if (waliData[i][3] === username && waliData[i][4] === password) {
      return { role: 'wali_kelas', name: waliData[i][1], id: waliData[i][0] };
    }
  }
  
  return { status: "error", message: "Username atau Password salah" };
}

/**
 * API ENDPOINT (doPost)
 */
function doPost(e) {
  const params = JSON.parse(e.postData.contents);
  let result;
  
  switch(params.action) {
    case 'login':
      result = checkLogin(params.username, params.password);
      break;
    case 'init':
      result = initDatabase();
      break;
    default:
      result = { status: "error", message: "Aksi tidak dikenal" };
  }
  
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}
